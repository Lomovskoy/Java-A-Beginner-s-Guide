									/********************************
									 *								*
									 *	 Наиболее употребительные	*
									 *	методы из класса Throwable	*								 
									 *								*
									 ********************************/
+---------------------------------------+-----------------------------------------------------------------------+
| 			Метод						|							Описание									|
+---------------------------------------+-----------------------------------------------------------------------+
|  Throwable fillInStackTrace()			|	Возвращает обьект Throwable, солдержащий полную рассировку стека 	|
|										|	исключение. Этот объект пригодин для повторного 					|
|										|	генерирования исключений.											|
+---------------------------------------+-----------------------------------------------------------------------+
|  String getLocalizedMessage()	 		|	Возвращает описаине осключения, 									|
|										|	локализованное по региональным стандартам.							|
+---------------------------------------+-----------------------------------------------------------------------+
|  String getMessage()			 		|	Возвращает описаине осключения.										|
+---------------------------------------+-----------------------------------------------------------------------+
|  void printStackTrace()				|	Выводит трассировку стека исключений.								|
+---------------------------------------+-----------------------------------------------------------------------+
|  void printStackTrace				 	|	Выводит трассировку стека исключений в указанный поток.				|						|
|  (PrintStream поток)					|																		|
+---------------------------------------+-----------------------------------------------------------------------+
|  void printStackTrace					|	Направляет трассировку стека исключений в указанный поток.			|
|  (PrintWriter поток)					|																		|
+---------------------------------------+-----------------------------------------------------------------------+
|  String toString()			 		|	Возвращает объект типа String, содержащий полное описание 			|
|										|	исключения. Этот метод вызывается из метода println() 				|
|										|	при выводе объекта типа Throwable									|
+---------------------------------------+-----------------------------------------------------------------------+
